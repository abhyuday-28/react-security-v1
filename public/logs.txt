import os

# Function to search for a byte sequence in memory
def memsrch(memory, pattern):
    for i in range(len(memory) - len(pattern) + 1):
        if memory[i:i+len(pattern)] == pattern:
            return i
    return -1

# Function to check if a file exists and get its size
def get_file_size(file_path):
    if os.path.isfile(file_path):
        return os.path.getsize(file_path)
    return -1
